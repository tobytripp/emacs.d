** Js-Mode

   #+BEGIN_SRC emacs-lisp
     (require-package 'js-comint)
     (defun toby/js-init ()
       (glasses-mode t)
       (setq mode-name "Js"))

     (eval-after-load "js-mode"
       '(progn
          (add-hook 'js-mode-hook 'toby/js-init)
          (require 'js-comint)))

     (defun jslint-thisfile ()
       (interactive)
       (compile (format "jsl -process %s" (buffer-file-name))))
   #+END_SRC

(provide 'javascript-init)
